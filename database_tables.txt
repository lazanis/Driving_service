# TABLES
Create table USERS
(
    id varchar(50) primary key,
    name varchar(50),
    surname varchar(50),
    role varchar(10),
    date_of_birth bigint,
    username varchar(50),
    pwd varchar(50),
    email varchar(50)
)

Create table CARS
(
    id varchar(50) primary key,
    user_id varchar(50),
    seats int,
    type varchar(50),
    foreign key (user_id) references USERS(id)
)
